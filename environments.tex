% global settings
\tcbset{%
    parbox = false,
}

% styles
\tcbset{%
    basestyle/.style={%
        enhanced,
        sharp corners,
        coltitle=black,
        colback=white,
        fonttitle=\sffamily\bfseries,
        description font=\mdseries,
        description delimiters parenthesis,
        seperator sign none,
    }
    boxstyle/.style = {%
        basestyle,
        colframe = black,
        after title = {.},
        attach boxed title to top left = {xshift=2mm, yshift=-2.5mm},
        boxed title style = {%
            colframe = white,
            colback = white,
            sharp corners,
        },
        boxrule = 0.5pt,
    },
    plainstyle/.style = {%
        basestyle,
        boxrule = 0pt,
        boxsep = 0mm,
        left = 0mm,
        breakable,
        attach title to upper = \;,
        after title = {.},
        colframe = white,
    },
}
